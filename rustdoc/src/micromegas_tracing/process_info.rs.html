1//! Process metadata
2use micromegas_transit::uuid_utils;
3use std::collections::HashMap;
4
5use serde::{Deserialize, Serialize};
6#[derive(Debug, Clone, Serialize, Deserialize)]
7pub struct ProcessInfo {
8    #[serde(
9        deserialize_with = "uuid_utils::uuid_from_string",
10        serialize_with = "uuid_utils::uuid_to_string"
11    )]
12    pub process_id: uuid::Uuid,
13    pub exe: String,
14    pub username: String,
15    pub realname: String,
16    pub computer: String,
17    pub distro: String,
18    pub cpu_brand: String,
19    pub tsc_frequency: i64,
20    /// RFC 3339
21    pub start_time: chrono::DateTime<chrono::Utc>,
22    pub start_ticks: i64,
23    #[serde(
24        deserialize_with = "uuid_utils::opt_uuid_from_string",
25        serialize_with = "uuid_utils::opt_uuid_to_string"
26    )]
27    pub parent_process_id: Option<uuid::Uuid>,
28    pub properties: HashMap<String, String>,
29}
