6/// Query all processes ordered by last update time
7pub async fn query_all_processes(client: &mut Client) -> Result<Vec<RecordBatch>> {
8    let sql = "SELECT process_id, start_time, last_update_time, exe, computer, username, cpu_brand, distro, properties
9               FROM processes
10               ORDER BY last_update_time DESC";
11    client.query(sql.to_owned(), None).await
12}

14/// Query log entries for a process with optional level filter
15pub async fn query_log_entries(
16    client: &mut Client,
17    process_id: &str,
18    level_filter: Option<&str>,
19    limit: usize,
20) -> Result<FlightRecordBatchStream> {
21    let level_condition = match level_filter {
22        Some("fatal") => "AND level = 1",
23        Some("error") => "AND level = 2",
24        Some("warn") => "AND level = 3",
25        Some("info") => "AND level = 4",
26        Some("debug") => "AND level = 5",
27        Some("trace") => "AND level = 6",
28        _ => "",
29    };
30
31    let sql = format!(
32        "SELECT time, level, target, msg
33         FROM log_entries
34         WHERE process_id = '{}'
35         {}
36         ORDER BY time DESC
37         LIMIT {}",
38        process_id, level_condition, limit
39    );
40
41    client.query_stream(sql, None).await
42}

44/// Query process statistics (log entries, measures, trace events, thread count)
45pub async fn query_process_statistics(
46    client: &mut Client,
47    process_id: &str,
48) -> Result<Vec<RecordBatch>> {
49    let sql = format!(
50        "SELECT
51            SUM(CASE WHEN array_has(\"streams.tags\", 'log') THEN nb_objects ELSE 0 END) as log_entries,
52            SUM(CASE WHEN array_has(\"streams.tags\", 'metrics') THEN nb_objects ELSE 0 END) as measures,
53            SUM(CASE WHEN array_has(\"streams.tags\", 'cpu') THEN nb_objects ELSE 0 END) as trace_events,
54            COUNT(DISTINCT CASE WHEN array_has(\"streams.tags\", 'cpu') THEN stream_id ELSE NULL END) as thread_count
55         FROM blocks
56         WHERE process_id = '{}'",
57        process_id
58    );
59
60    client.query(sql, None).await
61}
