10/// Span types to include in the trace
11pub enum SpanTypes {
12    Thread,
13    Async,
14    Both,
15}
16
17impl SpanTypes {
18    fn as_str(&self) -> &'static str {
19        match self {
20            SpanTypes::Thread => "thread",
21            SpanTypes::Async => "async",
22            SpanTypes::Both => "both",
23        }
24    }
25}

27/// Formats a Perfetto trace with configurable span types using server-side perfetto_trace_chunks function.
28///
29/// This function queries the FlightSQL server using the perfetto_trace_chunks table function
30/// which generates Perfetto trace data server-side and streams it back as binary chunks.
31///
32/// # Arguments
33/// * `span_types` - Types of spans to include: Thread, Async, or Both
34pub async fn format_perfetto_trace(
35    client: &mut Client,
36    process_id: &str,
37    query_range: TimeRange,
38    span_types: SpanTypes,
39) -> Result<Vec<u8>> {
40    // Use the perfetto_trace_chunks table function to get binary chunks
41    // Note: ORDER BY not needed since chunks are naturally produced in order (0, 1, 2, ...)
42    let sql = format!(
43        r#"
44        SELECT chunk_id, chunk_data
45        FROM perfetto_trace_chunks(
46            '{}',
47            '{}',
48            TIMESTAMP '{}',
49            TIMESTAMP '{}'
50        )
51        "#,
52        process_id,
53        span_types.as_str(),
54        query_range.begin.to_rfc3339(),
55        query_range.end.to_rfc3339()
56    );
57
58    let batches = client.query(sql, Some(query_range)).await?;
59
60    // Collect all chunks and reassemble them in order
61    let mut trace_data = Vec::new();
62    for batch in batches {
63        let chunk_data: &BinaryArray = typed_column_by_name(&batch, "chunk_data")?;
64
65        // Chunks are already in order from server-side generation
66        for i in 0..batch.num_rows() {
67            let chunk = chunk_data.value(i);
68            trace_data.extend_from_slice(chunk);
69        }
70    }
71
72    if trace_data.is_empty() {
73        anyhow::bail!("No trace data generated for process {}", process_id);
74    }
75
76    Ok(trace_data)
77}

79/// Writes a Perfetto trace to a file with configurable span types.
80///
81/// This function generates traces with thread spans, async spans, or both.
82///
83/// # Arguments
84/// * `span_types` - Types of spans to include: Thread, Async, or Both
85pub async fn write_perfetto_trace(
86    client: &mut Client,
87    process_id: &str,
88    begin: DateTime<Utc>,
89    end: DateTime<Utc>,
90    out_filename: &str,
91    span_types: SpanTypes,
92) -> Result<()> {
93    let buf =
94        format_perfetto_trace(client, process_id, TimeRange::new(begin, end), span_types).await?;
95    let mut file = File::create(out_filename).await?;
96    file.write_all(&buf).await?;
97    Ok(())
98}
