21#[derive(Error, Debug)]
22pub enum GatewayError {
23    #[error("Internal server error: {0}")]
24    Internal(#[from] anyhow::Error),
25}
26
27impl IntoResponse for GatewayError {
28    fn into_response(self) -> Response<Body> {
29        let (status, message) = match &self {
30            GatewayError::Internal(err) => {
31                let msg = format!("{err:?}");
32                (StatusCode::INTERNAL_SERVER_ERROR, msg)
33            }
34        };
35        (status, message).into_response()
36    }
37}

39#[derive(Debug, Deserialize)]
40pub struct QueryRequest {
41    sql: String,
42}
43
44pub async fn handle_query(
45    headers: HeaderMap,
46    Json(request): Json<QueryRequest>,
47) -> Result<String, GatewayError> {
48    info!("request={request:?}");
49    let flight_url = std::env::var("MICROMEGAS_FLIGHTSQL_URL")
50        .with_context(|| "error reading MICROMEGAS_FLIGHTSQL_URL environment variable")?
51        .parse::<Uri>()
52        .with_context(|| "parsing flightsql url")?;
53    let tls_config = ClientTlsConfig::new().with_native_roots();
54    let channel = Channel::builder(flight_url)
55        .tls_config(tls_config)
56        .with_context(|| "tls_config")?
57        .connect()
58        .await
59        .with_context(|| "connecting grpc channel")?;
60    let mut client = Client::new(channel);
61    if let Some(auth_header) = headers.get(AUTHORIZATION) {
62        client.inner_mut().set_header(
63            AUTHORIZATION.as_str(),
64            auth_header
65                .to_str()
66                .with_context(|| "converting auth header to a string")?,
67        );
68    }
69    let batches = client.query(request.sql, None).await?;
70    if batches.is_empty() {
71        return Ok("[]".to_string());
72    }
73
74    let mut buffer = Vec::new();
75    let mut json_writer = Writer::<_, JsonArray>::new(&mut buffer);
76    let batch_refs: Vec<&RecordBatch> = batches.iter().collect();
77    json_writer
78        .write_batches(&batch_refs)
79        .with_context(|| "json_writer.write_batches")?;
80    json_writer.finish().unwrap();
81    Ok(String::from_utf8(buffer).with_context(|| "converting json buffer to utf8")?)
82}
83
84pub fn register_routes(router: Router) -> Router {
85    router.route("/gateway/query", post(handle_query))
