1//! Top-level procedural macros for micromegas
2//!
3//! This crate provides high-level procedural macros that integrate multiple
4//! micromegas components for a seamless developer experience.
5
6use quote::quote;
7use syn::{AttributeArgs, ItemFn, Lit, Meta, NestedMeta, parse_macro_input};
8
9/// micromegas_main: Creates a tokio runtime with proper micromegas tracing callbacks and telemetry setup
10///
11/// This is a drop-in replacement for `#[tokio::main]` that automatically configures:
12/// - Tokio runtime with proper micromegas tracing thread lifecycle callbacks
13/// - Telemetry guard with sensible defaults (ctrl-c handling, debug level)
14///
15/// # Parameters
16///
17/// - `interop_max_level`: Optional interop max level override (e.g., "info", "debug", "warn")
18/// - `max_level_override`: Optional max level override (e.g., "info", "debug", "warn")
19///
20/// # Examples
21///
22/// ```ignore
23/// use micromegas::tracing::prelude::*;
24///
25/// #[micromegas_main]
26/// async fn main() -> Result<(), Box<dyn std::error::Error>> {
27///     info!("Server starting - telemetry already configured!");
28///     Ok(())
29/// }
30///
31/// #[micromegas_main(interop_max_level = "info")]
32/// async fn main() -> Result<(), Box<dyn std::error::Error>> {
33///     info!("Server starting with info interop level!");
34///     Ok(())
35/// }
36///
37/// #[micromegas_main(max_level_override = "warn", interop_max_level = "info")]
38/// async fn main() -> Result<(), Box<dyn std::error::Error>> {
39///     info!("Server starting with both level overrides!");
40///     Ok(())
41/// }
42/// ```
43#[proc_macro_attribute]
44pub fn micromegas_main(
45    args: proc_macro::TokenStream,
46    input: proc_macro::TokenStream,
47) -> proc_macro::TokenStream {
48    let args = parse_macro_input!(args as AttributeArgs);
49    let function = parse_macro_input!(input as ItemFn);
50
51    // Ensure the function is async and named main
52    if function.sig.asyncness.is_none() {
53        panic!("micromegas_main can only be applied to async functions");
54    }
55
56    if function.sig.ident != "main" {
57        panic!("micromegas_main can only be applied to the main function");
58    }
59
60    // Parse the level override parameters if provided
61    let mut interop_max_level: Option<String> = None;
62    let mut max_level_override: Option<String> = None;
63
64    for arg in args {
65        match arg {
66            NestedMeta::Meta(Meta::NameValue(nv)) if nv.path.is_ident("interop_max_level") => {
67                if let Lit::Str(lit_str) = nv.lit {
68                    interop_max_level = Some(lit_str.value());
69                } else {
70                    panic!("interop_max_level must be a string literal");
71                }
72            }
73            NestedMeta::Meta(Meta::NameValue(nv)) if nv.path.is_ident("max_level_override") => {
74                if let Lit::Str(lit_str) = nv.lit {
75                    max_level_override = Some(lit_str.value());
76                } else {
77                    panic!("max_level_override must be a string literal");
78                }
79            }
80            _ => panic!(
81                "Unsupported attribute argument. Supported: interop_max_level, max_level_override"
82            ),
83        }
84    }
85
86    let original_block = &function.block;
87    let return_type = &function.sig.output;
88
89    // Helper function to convert level string to LevelFilter token
90    let level_to_filter = |level: &str| -> proc_macro2::TokenStream {
91        match level.to_lowercase().as_str() {
92            "trace" => quote! { micromegas::tracing::levels::LevelFilter::Trace },
93            "debug" => quote! { micromegas::tracing::levels::LevelFilter::Debug },
94            "info" => quote! { micromegas::tracing::levels::LevelFilter::Info },
95            "warn" => quote! { micromegas::tracing::levels::LevelFilter::Warn },
96            "error" => quote! { micromegas::tracing::levels::LevelFilter::Error },
97            "off" => quote! { micromegas::tracing::levels::LevelFilter::Off },
98            _ => {
99                panic!("Invalid level value. Must be one of: trace, debug, info, warn, error, off")
100            }
101        }
102    };
103
104    // Generate the telemetry guard builder with optional level overrides
105    let mut builder_calls = vec![
106        quote! { .with_ctrlc_handling() },
107        quote! { .with_local_sink_max_level(micromegas::tracing::levels::LevelFilter::Debug) },
108        quote! { .with_process_property("version".to_string(), env!("CARGO_PKG_VERSION").to_string()) },
109    ];
110
111    if let Some(level) = max_level_override {
112        let level_filter = level_to_filter(&level);
113        builder_calls.push(quote! { .with_max_level_override(#level_filter) });
114    }
115
116    if let Some(level) = interop_max_level {
117        let level_filter = level_to_filter(&level);
118        builder_calls.push(quote! { .with_interop_max_level_override(#level_filter) });
119    }
120
121    let telemetry_guard_builder = quote! {
122        micromegas::telemetry_sink::TelemetryGuardBuilder::default()
123            #(#builder_calls)*
124            .build()
125    };
126
127    let expanded = quote! {
128        fn main() #return_type {
129            // Check CPU tracing setting before building runtime
130            let cpu_tracing_enabled = std::env::var("MICROMEGAS_ENABLE_CPU_TRACING")
131                .map(|v| v == "true")
132                .unwrap_or(false); // Default to disabled for minimal overhead
133
134            // Set up telemetry guard BEFORE building tokio runtime
135            // This ensures dispatch is initialized before worker threads start
136            let _telemetry_guard = #telemetry_guard_builder;
137
138            // Build the runtime with conditional tracing callbacks
139            let runtime = {
140                use micromegas::tracing::runtime::TracingRuntimeExt;
141                let mut builder = tokio::runtime::Builder::new_multi_thread();
142                builder.enable_all();
143                builder.thread_name(env!("CARGO_PKG_NAME"));
144                if cpu_tracing_enabled {
145                    builder.with_tracing_callbacks();
146                }
147                builder.build().expect("Failed to build tokio runtime")
148            };
149
150            runtime.block_on(async move {
151                // Execute the original main function body
152                #original_block
153            })
154        }
155    };
156
157    proc_macro::TokenStream::from(expanded)
158}
